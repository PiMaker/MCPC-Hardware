// Debug clocking
wire [7:0] c1_out;
wire [7:0] c2_out;
wire [7:0] c3_out;

counter c1 (
	.out(c1_out),
	.enable(1'b1),
	.clk(MAX10_CLK1_50),
	.reset(RST)
);

counter c2 (
	.out(c2_out),
	.enable(1'b1),
	.clk(c1_out[7]),
	.reset(RST)
);

counter c3 (
	.out(c3_out),
	.enable(1'b1),
	.clk(c2_out[7]),
	.reset(RST)
);

reg act;

always @(posedge c3_out[1])
begin
	if (act) begin
		fb_addr_bus <= fb_addr_bus + 1;
		fb_data_bus <= 8'h42;
	end else begin
		fb_data_bus <= 8'h0;
	end
	
	act <= ~act;
end
